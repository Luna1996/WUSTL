	\begin{tlist}{2}
		\item[$\bullet$] $ab$ is distinguishable with $\Lambda,a,aa$ by $\Lambda$.
		\item[$\bullet$] $a$ is distinguishable with $\Lambda,aa$ by $b$.
		\item[$\bullet$] $\Lambda$ is distinguishable with $aa$ by $ab$.
	\end{tlist}
